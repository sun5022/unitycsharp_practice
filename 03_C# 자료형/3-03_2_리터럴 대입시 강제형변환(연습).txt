//한 정수변수에 리터럴 2.1f 를 강제 형변환하여 대입하고 출력한후
//실수 변수에 실수 리터럴을 대입하고 출력
// 변수 num선언
// 변수 num에 2.1f를 정수로 강제형변환하여 대입
// 변수 num 출력
// 실수 fNum 선언
// 실수 fNum에 2.1f 대입
// fNum 출력
















































// 변수 num선언
int num;
// 변수 num에 2.1f를 정수로 강제형변환하여 대입
num = (int)2.1f;
// 변수 num 출력
print(num);
// 실수 fNum 선언
float fNum;
// 실수 fNum에 2.1f 대입
fNum = 2.1f;
// fNum 출력
print(fNum);
