// 변수 num선언
// 변수 num에 10 대입
// 변수 num 출력
// 실수 fNum 선언
// 실수 fNum에 10.2 대입
// fNum 출력

















































// 변수 num선언
int num;
// 변수 num에 10 대입
num = 10;
// 변수 num 출력
print(num);
// 실수 fNum 선언
float fNum;
// 실수 fNum에 10.2 대입
fNum = 10.2f;
// fNum 출력
print(fNum);
